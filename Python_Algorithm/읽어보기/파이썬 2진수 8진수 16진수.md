# 파이썬 2진수 8진수 16진수

```python
>>> bin(42)
'0b101010'
>>> oct(42)
'0o52'
>>> hex(42)
'0x2a'

>>> bin(0b101010)
'0b101010'
>>> oct(0b101010)
'0o52'
>>> hex(0b101010)
'0x2a'
>>> str(0b101010)
'42'

>>> int(0o52)
42
>>> int(0b10)
2

int 쓸 때 문자열로 넣으려면
>>> int("0xa", 16)
10
>>> int("0o12", 8)
10
>>> int("0b1010", 2)
10

>>> format(42, 'b')
'101010'
>>> format(42, 'o')
'52'
>>> format(42, 'x')
'2a'
>>> format(42, 'X')
'2A'
>>> format(42, 'd')
'42'

이것도 나름 유용
```
